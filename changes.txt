+

+        self.base.cTrav = CollisionTraverser()
+

+        if is_down(KeyboardButton.up()):

+
+        if is_down(KeyboardButton.down()):
+            speed += 500.0
+
+        if is_down(KeyboardButton.left()):

+        dt = ClockObject.getGlobalClock().getDt()

+        # Animate movement
